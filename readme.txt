clone command-- cloning a repository on our local machine(means the laptop or pc)
SYNTAX- git clone <-some link->

status command-- displays the state of the code(types-- untracked, staged, modified, unmodified)
SYNTAX- git status

cd- change directory

ls- to list all the files of current directory

ls-a or ls -hidden -- to list all the hidden files of the directory

add command-- add new files in the working directory in the git-hub area
syntax-- git add <-add name->

commit command-- record all the changes made in the directory
syntax-- git commit -m "some message"
